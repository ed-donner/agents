# Python program to calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ...
# and multiply the total by 4.

terms = 10000
total = 0

for n in range(terms):
    if n % 2 == 0:
        total += 1 / (2 * n + 1)  # Adding the term
    else:
        total -= 1 / (2 * n + 1)  # Subtracting the term

result = total * 4  # Multiply the total by 4
print(result)
