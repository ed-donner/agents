```python
# Calculate the first 10,000 terms of the series and multiply the result by 4
def calculate_series(n_terms):
    total = 0
    for n in range(n_terms):
        term = (-1)**n / (2*n + 1)  # Calculate each term, using the formula for odd numbers
        total += term
    return total * 4  # Multiply the result by 4

# Calculate the first 10,000 terms
result = calculate_series(10000)
print(result)
```
