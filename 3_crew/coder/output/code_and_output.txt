```python
# Calculate the first 10,000 terms of the series
total = 0.0
for i in range(10000):
    term = 1 / (2 * i + 1)  # Series term calculation
    if i % 2 == 0:
        total += term  # add for even index
    else:
        total -= term  # subtract for odd index

final_result = total * 4  # Multiply the total by 4
print(final_result)
```
