```python
def calculate_series(num_terms=10000):
    """
    Calculates the sum of the first num_terms of the series:
    1 - 1/3 + 1/5 - 1/7 + ...
    and multiplies the total by 4.
    """
    total_sum = 0.0
    for i in range(num_terms):
        # Calculate the ith term using the formula: (-1)^i / (2*i + 1)
        term = ((-1)**i) / (2 * i + 1)
        total_sum += term
    
    # The series approximates pi/4, so we multiply by 4
    final_result = total_sum * 4
    return final_result

# Calculate the result for 10,000 terms
result = calculate_series(10000)

print("Result for 10,000 terms:")
print(result)
```
