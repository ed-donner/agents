# Python Program to calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ... and multiply by 4

def calculate_series(terms):
    total = 0.0
    for n in range(terms):
        total += ((-1) ** n) / (2 * n + 1)
    return total * 4

result = calculate_series(10000)
print(result)

# Output of the code
# 3.1414926535900345
