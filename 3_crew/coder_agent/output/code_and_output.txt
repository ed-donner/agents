```python
# Calculate the value of the series 1 - 1/3 + 1/5 - 1/7 + ... for 10,000 terms

# Initialize total
total = 0.0

# Calculate the series
for i in range(10000):
    # Determine the denominator according to the series pattern
    denominator = 2 * i + 1
    # Add or subtract the term based on whether the index is even or odd
    if i % 2 == 0:
        total += 1 / denominator  # Add if even
    else:
        total -= 1 / denominator  # Subtract if odd

# Multiply the total by 4 as per the requirement
total *= 4

# Print the final result
print("The total after calculating the series (first 10,000 terms) multiplied by 4 is:", total)

# Output will be stored in a text file
with open('output.txt', 'w') as f:
    f.write("The total after calculating the series (first 10,000 terms) multiplied by 4 is: {}\n".format(total))
```
