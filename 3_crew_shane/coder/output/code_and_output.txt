```python
# Program to calculate the first 10,000 terms of the series: 
# 1 - 1/3 + 1/5 - 1/7 + ... and multiply by 4

# Initialize the sum
series_sum = 0

# Calculate the series
for n in range(10000):
    term = 1 / (2 * n + 1)  # Calculate the term for 1/(2n+1)
    if n % 2 == 0:  # Add if n is even, subtract if n is odd
        series_sum += term
    else:
        series_sum -= term

# Multiply the sum by 4
result = series_sum * 4

# Prepare output for saving
output = f'The result of the series after 10,000 terms multiplied by 4 is: {result}\nNumber of iterations: 10000\n'

# Save the code and output to a text file
with open('series_output.txt', 'w') as f:
    f.write('# Program to calculate the first 10,000 terms of the series:\n')
    f.write('# 1 - 1/3 + 1/5 - 1/7 + ... and multiply by 4\n\n')
    f.write('# Initialize the sum\nseries_sum = 0\n\n')
    f.write('# Calculate the series\nfor n in range(10000):\n')
    f.write('    term = 1 / (2 * n + 1)  # Calculate the term for 1/(2n+1)\n')
    f.write('    if n % 2 == 0:  # Add if n is even, subtract if n is odd\n')
    f.write('        series_sum += term\n    else:\n        series_sum -= term\n\n')
    f.write('# Multiply the sum by 4\nresult = series_sum * 4\n\n')
    f.write('# Print the final result\nprint(result)\n')
    f.write(output)

# Print the final result
print(result)
```
