# Calculate the first 10,000 terms of the series: 1 - 1/3 + 1/5 - 1/7 + ...
# Multiply the total by 4

num_terms = 10000
sum_series = 0.0

for i in range(num_terms):
    term = (-1) ** i / (2 * i + 1)
    sum_series += term

result = 4 * sum_series

print(f"Calculated value for first {num_terms} terms (multiplied by 4): {result}")

# Output:
Calculated value for first 10000 terms (multiplied by 4): 3.1414926535900345
